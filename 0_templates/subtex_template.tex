\section{Measure Theory}

\subsection{Measurable Space}
\qquad A {\bf measurable space} is a set $\Omega$ with a {\bf $\sigma$-algebra} $\fu{F}\subseteq2^\Omega$, s.t. 
\begin{enumerate}
\item $\Omega\in\fu{F}$.
\item $A\in\fu{F}\implies\Omega\setminus A\in\fu{F}$.
\item $S\subseteq\fu{F},S\preceq\aleph_0\implies\bigcup S\in\fu{F}$.
\end{enumerate}
The {\bf $\sigma$-algebra generated by} $S\subseteq2^\Omega$, denoted by 
\[\sigma\inp{S},\]
is defined to be the intersection of all $\sigma$-algebra containing $S$\footnote{Notice $2^\Omega$ is a $\sigma$-algebra containing $S$, the definition is well-defined.}.
\begin{pro}
Let $\f{\Omega,\fu{F}}$ be a measurable space:
\begin{enumerate}
\item The definition is equivalent to replace (3) by:
\[S\subseteq\fu{F},S\preceq\aleph_0\implies\bigcap S\in\fu{F}.\]
\item $\oo\in\fu{F}$.
\item $A,B\in\fu{F}\implies A\setminus B\in\fu{F}$.
\end{enumerate}
\end{pro}
\qquad An {\bf algebra} is a collection $\fu{F}\subseteq2^\Omega$ s.t. 
\begin{enumerate}
\item $\Omega\in\fu{F}$.
\item $A\in\fu{F}\implies\Omega\setminus A\in\fu{F}$.
\item $S\subseteq\fu{F},S\prec\aleph_0\implies\bigcup S\in\fu{F}$.
\end{enumerate}
\begin{pro}
Let $\Omega$ be a set, and let $\fu{F}\subseteq2^\Omega$ be an algebra:
\begin{enumerate}
\item The definition is equivalent to replace (3) by:
\[S\subseteq\fu{F},S\prec\aleph_0\implies\bigcap S\in\fu{F}.\]
\item $\oo\in\fu{F}$.
\item $A,B\in\fu{F}\implies A\setminus B\in\fu{F}$.
\end{enumerate}
\end{pro}
\begin{lem}
Let $\fu{F}\subseteq2^\Omega$. Then $\fu{F}$ is a $\sigma$-algebra iff. $\fu{F}$ is an algebra and satisfies
\[\s{A_n}_{n\in\bb{N}}\subseteq\fu{F}\mbox{ increasing}\footnote{Which means: for every $n\in\bb{N}$, $A_n\subseteq A_{n+1}$.}\implies\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{F}.\]
\end{lem}
\qquad A {\bf $\pi$-system} is a collection $\fu{C}\subseteq2^\Omega$ s.t. 
\[S\subseteq\fu{C},S\prec\aleph_0\implies\bigcap S\in\fu{C}.\] 
A {\bf $\lambda$-system} is a collection $\fu{L}\subseteq2^\Omega$ s.t. 
\begin{enumerate}
\item $\Omega\in\fu{L}$.
\item $A,B\in\fu{L},A\subseteq B\implies B\setminus A\in\fu{L}$.
\item If $\s{A_n}_{n\in\bb{N}}\subseteq\fu{L}$ increasing, then $\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{L}$.
\end{enumerate}
The {\bf $\lambda$-system generated by} $S\subseteq2^\Omega$, denoted by 
\[\lambda\inp{S},\]
is defined to be the intersection of all $\lambda$-system containing $S$.
\begin{lem}[$\pi$-$\lambda$ Theorem]\label{plt1}
A collection $\fu{F}\subseteq2^\Omega$ is a $\pi$-system and a $\lambda$-system iff. $\fu{F}$ is a $\sigma$-algebra.
\end{lem}
\begin{co}\label{plt2}
If $\fu{C}$ is a $\pi$-system, then $\lambda\inp{\fu{C}}=\sigma\inp{\fu{C}}$.
\end{co}
\begin{p}
Every $\sigma$-algebra is a $\lambda$-system, we have $\lambda\inp{\fu{C}}\subseteq\sigma\inp{\fu{C}}$.\par 
It suffices to show $\lambda\inp{\fu{C}}$ is a $\sigma$-algebra, i.e. $\lambda\inp{\fu{C}}\supseteq\sigma\inp{\fu{C}}$.\par
Let 
\[\lambda_1\f{\fu{C}}=\s{A\in\lambda\inp{\fu{C}}|\forall B\in\fu{C}:A\cap B\in\lambda\inp{\fu{C}}},\]
and let
\[\lambda_2\f{\fu{C}}=\s{A\in\lambda\inp{\fu{C}}|\forall B\in\lambda\inp{\fu{C}}:A\cap B\in\lambda\inp{\fu{C}}},\]
then $\lambda_1\f{\fu{C}}$ and $\lambda_2\f{\fu{C}}$ are $\lambda$-systems containing $\fu{C}$, hence $\lambda\inp{\fu{C}}\subseteq\lambda_2\inp{\fu{C}}\subseteq\lambda_1\inp{\fu{C}}\subseteq\lambda\inp{\fu{C}}$. Therefore, $\lambda\inp{\fu{C}}=\lambda_2\inp{\fu{C}}$, $\lambda\inp{\fu{C}}$ is a $\pi$-system, by \plem{plt1}, $\lambda\inp{\fu{C}}$ is a $\sigma$-algebra.
\end{p}
\qquad The {\bf Borel $\sigma$-algebra} of a t.s. $\f{\Omega,\tau}$ is defined to be $\sigma\inp{\tau}$.






\subsection{Measure and Extension Theorem}
\qquad A {\bf semialgebra} is a collection $\fu{C}\subseteq2^\Omega$ s.t. 
\begin{enumerate}
\item $S\subseteq\fu{C},S\prec\aleph_0\implies\bigcap_\Omega S\in\fu{C}$.
\item $\forall A\in\fu{C},\exists S\subseteq\fu{C},0\prec S\prec\aleph_0$ pairwise disjoint\footnote{Which means: for every $i,j\in\bb{N}$, if $i\neq j$, then $A_i\cap A_j=\oo$.} s.t. $\Omega\setminus A=\bigcup S$.
\end{enumerate}
A {\bf measure} of a semialgebra $\fu{C}\subseteq2^\Omega$ is a function $\mu:\fu{C}\to[0,+\infty]$ s.t.
\begin{enumerate}
\item $\mu\f{\oo}=0$.
\item If $\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}$ pairwise disjoint, and if $\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{C}$, then
\[\mu\f{\bigcup\s{A_n}_{n\in\bb{N}}}=\sum_{i=0}^\infty\mu\f{A_i}.\]
\end{enumerate}
\begin{pro}
Let $\mu$ be a measure on a semialgebra $\fu{C}$:
\begin{enumerate}
\item If $A,B\in\fu{C}$, then $A\subseteq B\implies\mu\f{A}\leq\mu\f{B}$.
\item If $\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}$, and if $\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{C}$, then 
\[\mu\f{\bigcup\s{A_n}_{n\in\bb{N}}}\leq\sum_{i=0}^\infty\mu\f{A_i}.\]
\item If $\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}$ increasing, and if $\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{C}$, then 
\[\mu\f{\bigcup\s{A_n}_{n\in\bb{N}}}=\lim_{n\to\infty}\mu\f{A_n}.\]
\item If $\s{A_n}_{n=0}^k\subseteq\fu{C}$ with $\forall0\leq n\leq k:\mu\f{A_n}<\infty$, then 
\[\begin{array}{>{\displaystyle}l>{\displaystyle}l}
\mu\f{\bigcup\s{A_n}_{n=0}^k}=&\sum_{i=0}^k\mu\f{A_i}-\sum_{0\leq i<j\leq k}\mu\f{A_i\cap A_j}+\ldots\\
&+\f{-1}^k\mu\f{A_0\cap\ldots\cap A_k}
\end{array}.\]
\end{enumerate}
\end{pro}
\qquad The {\bf algebra generated by} a semialgebra $\fu{C}\subseteq2^\Omega$, denoted by
\[\fu{A}\inp{\fu{C}},\]
is defined to be the intersection of all algebra containing $\fu{C}$. 
\begin{lem}
\begin{enumerate}
\item[]
\item An algebra is a semialgebra.
\item Let $\fu{C}$ be a semialgebra, then
\[\fu{A}\inp{\fu{C}}=\s{A\in2^\Omega|\exists \s{B_i}_{i=0}^k\subseteq\fu{C}\mbox{ pairwise disjoint}:A=\bigcup \s{B_i}_{i=0}^k}.\]
\item $\sigma\inp{\fu{C}}=\sigma\inp{\fu{A}\inp{\fu{C}}}$.
\end{enumerate}
\end{lem}
\qquad The {\bf measure induced by} $\mu$ to the algebra $\fu{A}\inp{\fu{C}}$, denoted by $\bar{\mu}$, is defined by $\bar{\mu}:\fu{A}\inp{\fu{C}}\to[0,\infty]$:
\[\bar{\mu}\f{A}=\sum_{i=0}^k\mu\f{B_i},\]
where $\s{B_i}_{i=0}^k$ pairwise disjoint, and $A=\bigcup\s{B_i}_{i=0}^k$.
\begin{theo}
Let $\mu$ be a measure on a semialgebra $\fu{C}$. Then:
\begin{enumerate}
\item $\bar{\mu}$ is well-defined.
\item $\restr{\bar{\mu}}{\fu{C}}=\mu$.
\item $\bar{\mu}$ is a measure on $\fu{A}\inp{\fu{C}}$.
\end{enumerate}
\end{theo}
\begin{p}
\begin{enumerate}
\setcounter{enumi}{2}
\item[]
\item Let $\s{A_n}_{n\in\bb{N}}\subseteq\fu{A}$ be pairwise disjoint, then for each $n\in\bb{N}$, we have 
\[A_n=\bigcup\s{B^n_j|j\in\bb{N}:0\leq j\leq k_n\in\bb{N}},\] 
where $\s{B^n_j|j\in\bb{N}:0\leq j\leq k_n\in\bb{N}}\subseteq\fu{C}$ pairwise disjoint. Similarly,
\[\bigcup\s{A_n}_{n\in\bb{N}}=\bigcup\s{B_j|j\in\bb{N}:0\leq j\leq k\in\bb{N}},\] 
where $\s{B_j|j\in\bb{N}:0\leq j\leq k\in\bb{N}}\subseteq\fu{C}$ pairwise disjoint.
Therefore,
\[B_i=B_i\cap\bigcup\s{A_n}_{n\in\bb{N}}=\bigcup\s{A_n\cap B_i}_{n\in\bb{N}}=\bigcup\s{B_j^n\cap B_i}_{n\in\bb{N},0\leq j\leq k_n}.\]
and
\[\mu\f{B_i}=\sum_{n=0}^\infty\sum_{j=0}^{k_n}\mu\f{B_j^n\cap B_i},\]
which implies
\[\bar{\mu}\f{\bigcup\s{A_n}_{n\in\bb{N}}}=\sum_{i=0}^k\sum_{n=0}^\infty\sum_{j=0}^{k_n}\mu\f{B_j^n\cap B_i}=\sum_{n=0}^\infty\f{\sum_{i=0}^k\sum_{j=0}^{k_n}\mu\f{B_j^n\cap B_i}}.\]
While 
\[A_n=A_n\cap\bigcup\s{B_i}_{0\leq i\leq k}=\bigcup\s{B_j^n\cap B_i}_{0\leq i\leq k,0\leq j\leq k_n},\]
i.e.
\[\bar\mu\f{A_n}=\sum_{i=0}^k\sum_{j=0}^{k_n}\mu\f{B_j^n\cap B_i}.\]
Together,
\[\bar{\mu}\f{\bigcup\s{A_n}_{n\in\bb{N}}}=\sum_{n=0}^\infty\bar\mu\f{A_n}.\]
\end{enumerate}
\end{p}
\qquad The {\bf outer measure induced by} $\mu$, denoted by $\mu^\ast$, is defined by $\mu^\ast:2^\Omega\to[0,\infty]$:
\[\mu^\ast\f{A}=\inf\s{\sum_{n=0}^\infty\mu\f{A_n}|\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}:A\subseteq\bigcup\s{A_n}_{n\in\bb{N}}}.\]
A set $A\subseteq\Omega$ is {\bf $\mu^\ast$-measurable} iff.
\[\forall E\subseteq\Omega:\mu^\ast\f{E}=\mu^\ast\f{E\cap A}+\mu^\ast\f{E\cap\f{\Omega\setminus A}}.\]
The {\bf Carath\'{e}odory $\sigma$-algebra}, denoted by 
\[\fu{M}_{\mu^\ast},\] 
is the collection of all $\mu^\ast$-measurable sets.
\begin{theo}[Carath\'{e}odory Extension Theorem]
Let $\mu$ be a measure on a semialgebra $\fu{C}$:
\begin{enumerate}
\item $\restr{\mu^\ast}{\fu{A}\inp{\fu{C}}}=\bar{\mu}$.
\item If $\s{A_n}_{n\in\bb{N}}\subseteq2^\Omega$, then
\[\mu^\ast\f{\bigcup\s{A_n}_{n\in\bb{N}}}\leq\sum_{i=0}^\infty\mu^\ast\f{A_i}.\]
\item $\fu{C}\subseteq\fu{M}_{\mu^\ast}$.
\item $\fu{M}_{\mu^\ast}$ is a $\sigma$-algebra.
\item $\restr{\mu^\ast}{\fu{M}_{\mu^\ast}}$ is a measure.
\item $A\in2^\Omega,\mu^\ast\f{A}=0\implies2^A\subseteq\fu{M}_{\mu^\ast}$.
\end{enumerate}
\end{theo}
\begin{p}
\begin{enumerate}
\item[]
\item It suffices to show $\bar{\mu}\f{A}\leq\mu^\ast\f{A}$.\par
For all $\varepsilon>0$, there exists a collection $\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}$ s.t.
\[A\subseteq\bigcup\s{A_n}_{n\in\bb{N}},\qquad \sum_{n=0}^\infty\mu\f{A_n}\leq\mu^\ast\f{A}+\varepsilon.\]
Let $\s{B_i}_{i=0}^k$ pairwise disjoint s.t. $A=\bigcup\s{B_i}_{i=0}^k$, then 
\[A=A\cap\bigcup\s{A_n}_{n\in\bb{N}}=\bigcup\s{B_i\cap A_n}_{n\in\bb{N},0\leq i\leq k}.\]
Therefore,
\[\bar{\mu}\f{A}\leq\sum_{n=0}^\infty\sum_{i=0}^k\mu\f{B_i\cap A_n}=\sum_{n=0}^\infty\mu\f{A_n}\leq\mu^\ast\f{A}+\varepsilon.\]
Let $\varepsilon\to0$, we get $\bar{\mu}\f{A}\leq\mu^\ast\f{A}$.
\setcounter{enumi}{2}
\item We only show $\forall E\subseteq\Omega,\forall A\in\fu{C}:\mu^\ast\f{E}\geq\mu^\ast\f{E\cap A}+\mu^\ast\f{E\cap\f{\Omega\setminus A}}$.\par
For all $\varepsilon>0$, there exists a collection $\s{A_n}_{n\in\bb{N}}\subseteq\fu{C}$ s.t. 
\[E\subseteq\bigcup\s{A_n}_{n\in\bb{N}},\qquad \sum_{n=0}^\infty\mu\f{A_n}\leq\mu^\ast\f{E}+\varepsilon.\] 
Let $\s{B_i}_{i=0}^k\subseteq\fu{C}$ pairwise disjoint s.t. $\Omega\setminus A=\bigcup\s{B_i}_{i=0}^k$. Then
\[E\cap A\subseteq\bigcup\s{A_n\cap A}_{n\in\bb{N}},\qquad E\cap\f{\Omega\setminus A}\subseteq\bigcup\s{A_n\cap B_i}_{n\in\bb{N},0\leq i\leq k}.\]
Hence 
\[\begin{array}{>{\displaystyle}l>{\displaystyle}l}
\mu^\ast\f{E\cap A}+\mu^\ast\f{E\cap\f{\Omega\setminus A}}&\leq\sum_{n=0}^\infty\f{\mu\f{A_n\cap A}+\sum_{i=0}^k\mu\f{A_n\cap B_i}}\\
&=\sum_{n=0}^\infty\mu\f{A_n}\leq\mu^\ast\f{E}+\varepsilon
\end{array}.\]
Let $\varepsilon\to0$, we get $\mu^\ast\f{E}\geq\mu^\ast\f{E\cap A}+\mu^\ast\f{E\cap\f{\Omega\setminus A}}$.
\item We first show that $\fu{M}_{\mu^\ast}$ is an algebra:\par
For convenience, we write $A^c=\Omega\setminus A$:
\[\begin{array}{>{\displaystyle}r>{\displaystyle}l}
\mu^\ast\f{E}=&\mu^\ast\f{E\cap A_1}+\mu^\ast\f{E\cap A_1^c}\\
=&\mu^\ast\f{E\cap A_1\cap A_2}+\mu^\ast\f{E\cap A_1\cap A_2^c}\\
&+\mu^\ast\f{E\cap A_1^c\cap A_2}+\mu^\ast\f{E\cap A_1^c\cap A_2^c}
\end{array}.\]
Notice that $A_1\cup A_2=\f{A_1\cap A_2}\cup\f{A_1^c\cap A_2}\cup\f{A_1\cap A_2^c}$, and $\f{A_1\cap A_2}^c=A_1^c\cap A_2^c$. We have 
\[\mu^\ast\f{E}\geq\mu^\ast\f{E\cap\f{A_1\cup A_2}}+\mu^\ast\f{E\cap\f{A_1\cup A_2}^c}.\]
Now, we show that if $\s{A_n}_{n\in\bb{N}}\subseteq\fu{M}_{\mu^\ast}$ increasing, then $\bigcup\s{A_n}_{n\in\bb{N}}\in\fu{M}_{\mu^\ast}$.\par
Let $\s{B_n}_{n\in\bb{N}}$ be defined by
\begin{itemize}
\item $B_0=A_0$,
\item $B_{n+1}=A_{n+1}\setminus A_n$.
\end{itemize}
Then $\s{B_n}_{n\in\bb{N}}\subseteq\fu{M}_{\mu^\ast}$ pairwise disjoint, with $\bigcup\s{A_n}_{n\in\bb{N}}=\bigcup\s{B_n}_{n\in\bb{N}}$.\par
For $n\geq1$:
\[\begin{array}{>{\displaystyle}r>{\displaystyle}l}
\mu^\ast\f{E}=&\mu^\ast\f{E\cap A_n}+\mu^\ast\f{E\cap A_n^c}\\
=&\mu^\ast\f{E\cap A_n\cap B_n}+\mu^\ast\f{E\cap A_n\cap B_n^c}+\mu^\ast\f{E\cap A_n^c}\\
=&\mu^\ast\f{E\cap B_n}+\mu^\ast\f{E\cap A_{n-1}}+\mu^\ast\f{E\cap A_n^c}\\
=&\sum_{i=0}^n\mu^\ast\f{E\cap B_i}+\mu^\ast\f{E\cap A_n^c}\\
\geq&\sum_{i=0}^n\mu^\ast\f{E\cap B_i}+\mu^\ast\f{E\cap\f{\bigcup\s{A_n}_{n\in\bb{N}}}^c}\\
\end{array}.\]
Let $n\to\infty$, we have 
\[\mu^\ast\f{E}\geq\mu^\ast\f{E\cap\bigcup\s{A_n}_{n\in\bb{N}}}+\mu^\ast\f{E\cap\f{\bigcup\s{A_n}_{n\in\bb{N}}}^c}.\]
\item Let $\s{B_n}_{n\in\bb{N}}\subseteq\fu{M}_{\mu^\ast}$ pairwise disjoint. Let $\s{A_j}_{j\in\bb{N}}$ be defined by $A_j=\bigcup\s{B_n}_{n=j}^\infty$:
\[\begin{array}{>{\displaystyle}r>{\displaystyle}l}
\mu^\ast\f{A_0}=&\mu^\ast\f{A_0\cap B_0}+\mu^\ast\f{A_0\cap B_0^c}\\
=&\mu^\ast\f{B_0}+\mu^\ast\f{A_1}\\
=&\mu^\ast\f{B_0}+\mu^\ast\f{B_1}+\mu^\ast\f{A_2}\\
=&\sum_{i=0}^n\mu^\ast\f{B_i}+\mu^\ast\f{A_{n+1}}\\
\geq&\sum_{i=0}^n\mu^\ast\f{B_i}\\
\end{array}.\]
Let $n\to\infty$, we have 
\[\mu^\ast\f{\bigcup\s{B_n}_{n\in\bb{N}}}\geq\sum_{i=0}^\infty\mu^\ast\f{B_i}.\]
\item Let $B\in2^A$, then $\mu^\ast{E\cap B}=\mu^\ast\f{B}=\mu^\ast{A}=0$, and $\mu^\ast\geq\mu^\ast\f{E\cap B^c}$.
\end{enumerate}
\end{p}

\qquad A {\bf measure space} (meas.s.) is a measurable space $\f{\Omega,\fu{F}}$ with a measure on $\fu{F}$\footnote{Or a measure up to the restriction on $\fu{F}$}. The {\bf$\mu$-null subsets} of a m.s., denoted by $\fu{N}_\mu$, is defined by
\[\fu{N}_\mu=\s{A\subseteq\Omega|\exists B\supseteq A:\mu\f{B}=0}.\]
